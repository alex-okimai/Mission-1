TABLES VBAP.

TYPES : BEGIN OF TY_DATA,
  
         WHERE(60) TYPE C,
  
        END OF TY_DATA.

DATA: IT_W1 TYPE STANDARD TABLE OF TY_DATA,
          
       WA_W1 TYPE TY_DATA.


DATA:  IT_VBAP TYPE STANDARD TABLE OF VBAP ,
      
       WA_VBAP  TYPE VBAP .


PARAMETERS R_1 AS CHECKBOX.

PARAMETERS P_MATNR like VBAP-MATNR DEFAULT 'M-01'.

SELECT-OPTIONS S_VBELN FOR VBAP-VBELN  DEFAULT '0000004969' to
'0000004979'.

SELECT-OPTIONS S_MATKL FOR VBAP-MATKL DEFAULT '002'.
             
WA_W1   = 'MATNR = P_MATNR '.


APPEND WA_W1 TO IT_W1.

IF  R_1  IS NOT INITIAL.
 
 CLear IT_W1.
  
WA_W1   = 'MATKL IN S_MATKL '.
  
APPEND WA_W1 TO IT_W1.
ENDIF.
  
WA_W1   = 'AND VBELN IN S_VBELN '.
  
APPEND WA_W1 TO IT_W1.

SELECT * INTO TABLE  IT_VBAP FROM VBAP
  WHERE (IT_W1).

if ( 1 = 2 ) OR ( 3 = 4 ).

ENDIF.
LOOP AT IT_VBAP  INTO WA_VBAP.
  
WRITE :/ WA_VBAP-VBELN,WA_VBAP-POSNR,WA_VBAP-MATNR, WA_VBAP-MATKL .
ENDLOOP.
